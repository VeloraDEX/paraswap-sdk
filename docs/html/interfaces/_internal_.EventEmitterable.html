Interface EventEmitterable<T>

An EventEmitterable behaves similar to an EventEmitter except provides async access to its methods.

An EventEmitter implements the observer pattern.

interface EventEmitterable<T> {
    addListener(event: T, listener: Listener): Promise<EventEmitterable<T>>;
    emit(event: T, ...args: any[]): Promise<boolean>;
    listenerCount(event?: T): Promise<number>;
    listeners(event?: T): Promise<Listener[]>;
    off(event: T, listener?: Listener): Promise<EventEmitterable<T>>;
    on(event: T, listener: Listener): Promise<EventEmitterable<T>>;
    once(event: T, listener: Listener): Promise<EventEmitterable<T>>;
    removeAllListeners(event?: T): Promise<EventEmitterable<T>>;
    removeListener(event: T, listener: Listener): Promise<EventEmitterable<T>>;
}
